2. heapq

https://docs.python.org/zh-cn/3/library/heapq.html

小顶堆，可以是元组，第一个东西是键值

  • 可以在需要小顶堆的地方用
  • 如https://leetcode.cn/problems/vvXgSW/ 剑指 Offer II 078. 合并排序链表open in new window
# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution(object):
    def mergeKLists(self, lists):
        """
        :type lists: List[ListNode]
        :rtype: ListNode
        """
        if len(lists)==0:
            return 
        if lists[0]==None and len(lists)==1:
            return 
        q = []
        for i in lists:
            if i != None:
                tmp_head = ListNode()
                tmp_head.next = i
                heapq.heappush(q, (i.val, tmp_head))
        res = ListNode()
        cur = res
        while(len(q)):
            tmp_head = heapq.heappop(q)[1]
            tmp_node = tmp_head.next
            tmp_head.next = tmp_head.next.next
            if tmp_head.next!=None:
                heapq.heappush(q, (tmp_head.next.val, tmp_head))
            tmp_node.next = None
            cur.next = tmp_node
            cur = cur.next

        return res.next
